\newcommand{\phulucchapter}[1]{%
    \iffirstappendix
        \firstappendixfalse
        \vspace{1em}
    \else
        \clearpage
    \fi
    \refstepcounter{chapter}%
    \setcounter{section}{0}%
    \setcounter{table}{0}%
    \setcounter{figure}{0}%
    \section*{Phụ lục \thechapter. #1}%
    \addcontentsline{toc}{section}{Phụ lục \thechapter. #1}%
}
